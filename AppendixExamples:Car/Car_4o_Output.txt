package SedanSystemModel {




part def Sedan {
attribute engine : Engine;
attribute transmission : Transmission;
attribute chassis : Chassis;
attribute infotainmentSystem : InfotainmentSystem;
attribute brakeSystem : BrakeSystem;
attribute steeringSystem : SteeringSystem;

}




part def InfotainmentSystem {

}

}

}

}

}

attribute frontLighting : LightType;

}
